              $uibModalInstance.close();

      function check(t,i) {
        if (t.check) {
          remTraje(t,i);
        } else {
          addTraje(t);

      }
 
      function addTraje(t) {
        vm.trajes_selecionados.push(t);
        t.check = 1;
      }
 
      function remTraje(t) {
        for (var i = 0; i < vm.trajes_selecionados.length; i++) {

            vm.trajes_selecionados.splice(i,1);
          }
        }
        t.check = 0;
      }        
 
      function limparPesqAvancada() {
        vm.consulta = {
