(function(){
    'use strict';
 
    angular
        .module('sis.reserva')
        .controller('ReqItensModalController', ReqItensModalController);
    ReqItensModalController.$inject = ['$q','Data', '$modalInstance','logger', '$scope','$filter', '$modal', 'ReqItemService','UtilsFunctions', 'EstoqueService'];
    /* @ngInject */
    function ReqItensModalController($q,Data, $modalInstance,logger, $scope, $filter, $modal, ReqItemService,UtilsFunctions, EstoqueService) {
        var vm = this;
        vm.title = 'Requisição';
        vm.requisicao = Data.requisicao;
        vm.itensReq = Data.itensReq;
        vm.secretarias = Data.secretarias;
        vm.estoque = [];
        vm.ok = ok;
        vm.cancel = cancel;
        vm.soma = UtilsFunctions.soma;
        vm.addItem = addItem;
        vm.remItem = remItem;
        vm.validarQt = validarQt;
        vm.salvar = salvar;
 
        //////////
        //activate();
 
        function activate() {
            var promises = [];
            return $q.all(promises).then(function() {
            });
        }
 
        function validarQt(data) {
            if (!data || data <= 0) {
                return 'A quantidade deve ser igual ou maior que 0.01';
            }
        }
 
        function remItem(i,item) {
            vm.itensReq.splice(i,1);
            if (item.id_ri) {
                ReqItemService.deletar(item);
            }
        }
 
        function addItem() {
            var data = {
                requisitar:true,
                avancado:true,
                secretarias:vm.secretarias,
                id_subcategoria:vm.requisicao.id_subcategoria,
                id_secretaria:vm.requisicao.id_secretaria,
                id_sdfc:vm.requisicao.id_sdfc
            };
            var modalInstance = $modal.open({
              templateUrl: 'app/sistema/estoque/templates/estoque.html',
              controller: 'EstoqueModalController',
              controllerAs: 'vm',
              size: 'lg',
              backdrop:'static',
              resolve: {
                Data: function () {
                  return data;
                }
              }              
            }); 
 
            modalInstance.result.then(function (data) {
                angular.forEach(data.itens, function(value, key){
                    var sdi = value.id_sdi;
                    var item = $filter('filter')(vm.itensReq,{
                      id_sdi:sdi
                    });
                    if (item.length === 0) {
                        vm.itensReq.push(ReqItemService.novoItem(vm.requisicao.id_requisicao,value));
                    }
                });
            });
        }
 
        function salvar(itens) {
            var DtTemp = UtilsFunctions.copiarObjecto(itens);
            var newItens = [];
            if (vm.requisicao.id_requisicao) {//se existir o id da requisição entrar no processo
                angular.forEach(DtTemp, function(value, key){
                    if (value.id_ri) {//se existir o id do item vai ser atualização
                        if (value.edit) {
                            ReqItemService.update(value).then(function(r){
                                if (r.status == "ok") {
                                    //logger
                                }
                            });
                        }
                    } else {//criar novo
                        value.id_requisicao = vm.requisicao.id_requisicao;
                        newItens.push(value);
                    }
                });
                //Salvando itens
                if (newItens.length > 0) {
                    ReqItemService.create(newItens);
                }
            }
        }
 
 
        function ok(data) {
            $modalInstance.close(data);
        }
        function cancel(){
        	$modalInstance.dismiss('cancel');
        }
 
    }
})();
