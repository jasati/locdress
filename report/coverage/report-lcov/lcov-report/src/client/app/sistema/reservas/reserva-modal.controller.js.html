    ReqModalController.$inject = ['config','$q','Data', '$modalInstance','logger', '$scope','$filter', '$modal', 'RequisicaoService','UtilsFunctions', 'ReqItemService','SituacaoService','UsuarioService','MenssagensService','ReqReport'];

    function ReqModalController(config,$q,Data, $modalInstance,logger, $scope, $filter, $modal, RequisicaoService, UtilsFunctions, ReqItemService, SituacaoService,UsuarioService,MenssagensService,ReqReport) {
        var vm = this;
        vm.title = 'Reserva';
        vm.novaReq = Data.gerarReq;
        vm.requisicoes = Data.requisicoes;
        vm.secretarias = Data.secretarias;
        vm.itensReq = Data.itensReq;
        vm.situacoes = [];
        vm.isCollapsed = false;
        vm.soma = UtilsFunctions.soma;
        vm.ok = ok;
        vm.cancel = cancel;
        vm.itens = itens;
        vm.enderecoSec = enderecoSec;
        vm.cancelarRec = cancelarRec;
        vm.salvar = salvar;
        vm.permissao = RequisicaoService.verificarPermissao;
        vm.concluir = concluir;
        vm.confirmImp = confirmImp;
        vm.sitAlterado = false;
        vm.pathImg = config.urlImagem;
        vm.dadosPref = RequisicaoService.dadosPref;
        vm.convDate = convDate;
        vm.printReq = printReq;
        vm.alterarSit = alterarSit;
        vm.deleteReq = deleteReq;
        vm.permRePrint = permRePrint;
        vm.popover = {
          templateUrl: 'app/sistema/requisicao/templates/cancelar.html',
          title: 'Cancelar',
          open:false
        };

        activate();
 
        function activate() {
            getItensRec();
            getSituacoes();
            if (Data.gerarReq) {
                var promises = [addNumReq()];
                return $q.all(promises).then(function() {
                    if (vm.requisicoes.length > 1) {
                        logger.info('Foram geradas '+vm.requisicoes.length+' requisições a partir dos itens selecionados. ');
                    } else {
                        logger.info('Foi gerada '+vm.requisicoes.length+' requisição a partir dos itens selecionados. ');
                    }
                });
            }
        }
 
        function convDate (date) {
          var dt = new Date(date);
          return dt;
        }
 
        function permRePrint(req) {
            req.impresso = 0;
            RequisicaoService.update(req);
        }
 
        function alterarSit(req,sit) {
            if (vm.permissao(19)) {
                req.id_sr = sit;
                if (sit === 2) {
                    req.id_usuario_lib = RequisicaoService.userLogado.id_usuario;
                    var dt = new Date();
                    req.data_lib = UtilsFunctions.formatData(dt,dt.getHours()+':'+dt.getMinutes()+':'+dt.getSeconds());
                }
                RequisicaoService.update(req).then(function(result){
                    if (result.status === "ok") {
                        if (RequisicaoService.userLogado.id_usuario!=req.id_usuario) {
                            var m = "";
                            switch (req.id_sr){
                                case 1:
                                    m = 'Pendente de Análise';
                                    break;
                                case 2:
                                    m = "Autorizado";
                                    break;
                                case 3:  
                                    m = "Não Autorizado";
                                    break;
                            }
                            var msg = {
                                usuario_destinatario:req.usuario,
                                id_usuario_para:req.id_usuario,
                                tipo_anexo:'R',
                                anexo:req.id_requisicao,
                                mensagem:"A Requisição Nº "+req.n_requisicao+", está "+m
                            };
                            var data = {mensagem:msg};
                            var modalInstance = $modal.open({
                                templateUrl: 'app/sistema/menssagens/templates/msg-confirma.html',
                                controller: 'MsgModalController',
                                controllerAs: 'vm',
                                size: '',
                                backdrop:'static',
                                resolve: {
                                Data: function () {
                                    return data;
                                    }
                                }
                            });
                            modalInstance.result.then(function () {
                                MenssagensService.create(msg);
                            });
                        }
                        
                    }
                });
            }
        }
        
        function getSituacoes() {
            var prm = {
                id_rs:"",
                incluir:"1,2,3"
            };
            SituacaoService.read(prm).then(function (data){
                vm.situacoes = data.reg;
            });
        }
 
        function concluir() {
            var c = false;
            if (vm.novaReq) {
                for (var i = 0; i < vm.requisicoes.length; i++) {
                    if (!vm.requisicoes[i].id_requisicao) {
                        c = true;
                    }
                }
            }
            return c;
        }
        function printReq(req) {
            if (req.id_sr === 2) {
                if (req.impresso && !vm.permissao(19)) {
                    logger.warning('Essa requisição já foi impressa, para re-imprimir entre em contato com o administrador das requisições.');
                } else {
                    if (req.id_usuario===RequisicaoService.userLogado.id_usuario) {
                        confirmImp(req);
                    }                    
                    var data = {requisicao:req,itensReq:vm.itensReq,modelo:'requisicao'};
                    var modalInstance = $modal.open({
                        templateUrl: 'app/sistema/requisicao/templates/requisicao-print.html',
                        controller: 'ReqPrintModalController',
                        controllerAs: 'vm',
                        size: 'lg',
                        backdrop:'static',
                        keyboard:false,
                        resolve: {
                        Data: function () {
                            return data;
                            }
                        }
                    });
                    modalInstance.result.then(function (req) {
 
                    });
                }
            } else {
                logger.warning('Você só poderá imprimir se a requisição estiver autorizada');
            }
        }
 
        function confirmImp(r) {
            r.impresso = 1;
            RequisicaoService.update(r);
        }
 
        function salvar(req) {
            if (req.id_requisicao === null) {
                //criar
                var enviarMsg = false;
                if (!vm.permissao(19)) {
                    enviarMsg = true;
                }
                var itens = req.itens;
                RequisicaoService.create(req).then(function(result) {
                    if (result.status == "ok") {
                        req.id_requisicao = result.last_insert;
                        for (var i = 0; i < itens.length; i++) {
                            itens[i].id_requisicao = result.last_insert;
                        }
                        ReqItemService.create(itens).then(function (result){
                            if (enviarMsg) {
                                var usuarioAdm = null;
                                UsuarioService.getUserModulo(19).then(function (user){
                                    if (user.reg.length > 0) {
                                        usuarioAdm = user.reg[0].id_usuario;
                                        var msg = {
                                            id_usuario_para:usuarioAdm,
                                            tipo_anexo:'R',
                                            anexo:req.id_requisicao,
                                            mensagem:"Pedido de Análise da Requisição Nº "+req.n_requisicao
                                        };
                                        MenssagensService.create(msg);
                                    } else {
                                        logger.warning("O sistema não encontrou um usuário administrador de requisições para análisar seu pedido.\n Entre em contato com o suporte.");
                                    }
                                });
                            }
                        });
 
                    }
                });
 
            } else {
                RequisicaoService.update(req);
            }
        }
 
        function itens(req) {
            var data = {requisicao:req,itensReq:req.itens,secretarias:vm.secretarias};
            var modalInstance = $modal.open({
                templateUrl: 'app/sistema/requisicao/templates/requisicao-itens.html',
                controller: 'ReqItensModalController',
                controllerAs: 'vm',
                size: 'lg',
                backdrop:'static',
                resolve: {
                Data: function () {
                    return data;
                    }
                }
            });
            modalInstance.result.then(function (save) {
 
            });
        }
        function addNumReq() {
            angular.forEach(vm.requisicoes, function(value, key){
                RequisicaoService.getNumDoc(1,value.id_secretaria).then(function(res){
                    value.n_requisicao = 'R'+res.result;
                });
            });
        }
        function enderecoSec(req) {
            if (!req.id_requisicao) {
                var end = '';
                for (var i = 0; i < vm.secretarias.length; i++) {
                    if (vm.secretarias[i].id_secretaria == req.id_secretaria) {
                        end = 'ENDEREÇO: '+vm.secretarias[i].endereco+'\n';
                        end = end+'BAIRRO: '+vm.secretarias[i].bairro+'\n';
                        end = end+'COMPLEMENTO: '+vm.secretarias[i].complemento+'\n';
                        break;
                    } 
                    
                }
            req.end_entrega = end;
            }
        }
        
 
 
        function getItensRec() {
            if (vm.requisicoes[0].itens === undefined) {
                ReqItemService.read(vm.requisicoes[0].id_requisicao).then(function(i) {
                    vm.requisicoes[0].itens = i.reg;
                    vm.itensReq = i.reg;
                });
            }
        }
 
        function cancelarRec(r,res) {
            if (res == 'S') {
                vm.requisicoes.splice(r,1);
                if (vm.requisicoes.length === 0) {
                    cancel();
                }
            }
        }
 
         function deleteReq(index) {
          if (vm.permissao(19)) {
            var data = {requisicao:index};
            var modalInstance = $modal.open({
              templateUrl: 'app/sistema/requisicao/templates/delete.html',
              controller: 'ReqDelModalController',
              controllerAs: 'vm',
              size: '',
              backdrop:'static',
              resolve: {
                Data: function () {
                  return data;
                }
              }              
            });
            
            modalInstance.result.then(function (del) {
              RequisicaoService.deletar(del).then(function (data) {
                if (data.status =="ok") {
                    ok('');
                }
              });
            });
          }
        }            
 
        function ok(data) {
            $modalInstance.close(data);
        }

        	$modalInstance.dismiss('cancel');
        }
