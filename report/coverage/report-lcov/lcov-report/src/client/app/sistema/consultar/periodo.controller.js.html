(function(){
    'use strict';
 
    angular
        .module('sis.consulta')
        .controller('PeriodoController', PeriodoController);
    PeriodoController.$inject = ['$uibModalInstance', '$scope'];
    /* @ngInject */
    function PeriodoController($uibModalInstance, $scope) {
        var vm = this;
        vm.title = 'Infome o Per√≠odo da Reserva';
        vm.caledario = {
            dataIni:'',
            dataFim:''
        };
        vm.icon = "fa-calendar";
 
        vm.ok = ok;
        vm.cancel = cancel;
 
        activate();
 
        function activate() {
            dataPadrao();
        }
 
        function dataPadrao() {
            var dataIni = new Date();
            var dataFim = new Date();
            dataFim.setDate(dataIni.getDate() + 5);
            vm.caledario.dataIni = dataIni;
            vm.caledario.dataFim = dataFim;
        }        
 
        function ok() {
            $uibModalInstance.close(vm.caledario);
        }
        function cancel(){
        	$uibModalInstance.dismiss('cancel');
        }
    }
})();
