(function(){
    'use strict';
 
    angular
        .module('sis.reserva')
        .controller('ReqPrintModalController', ReqPrintModalController);
    ReqPrintModalController.$inject = ['$q','Data', '$modalInstance','logger', '$scope','$filter', '$modal', 'RequisicaoService','config','ReqReport'];
    /* @ngInject */
    function ReqPrintModalController($q,Data, $modalInstance,logger, $scope, $filter, $modal, RequisicaoService,config,ReqReport ) {
        var vm = this;
        vm.title = 'Pré-Visualizar Impressão da Requisição';
        vm.requisicao = Data.requisicao;
        vm.itensReq = Data.itensReq;
        vm.pdf = '';
        vm.pdfName = '';
        vm.deletePdf = deletePdf;
 
 
        activate();
 
        function activate() {
            var promises = [];
            var page = '';
            var numero = 0;
            if (Data.modelo == 'requisicao') {
                promises = [ReqReport.getTplRequisicao(vm.requisicao,vm.itensReq)];
                page = 'R';
                numero = vm.requisicao.n_requisicao;
            } else if (Data.modelo == 'listagem') {
                var n = new Date();
                promises = [ReqReport.getTplListaReq(vm.requisicao)];
                page = 'L';
                numero = n.getTime();
            }
            $q.all(promises).then(function(html){
                createPdf(html[0],page,numero);
            });
        }
 
        function createPdf(html,page,numero) {
            var data = {
                prm:'C',
                html:html,
                nomePrefix:'req'+RequisicaoService.userLogado.id_usuario+'_',
                numero:numero,
                page:page//R retrato L Landscape
            };
            RequisicaoService.getPdf(data).then(function(result){
                vm.pdfName = result.report_name;
                vm.pdf = config.report+result.report_name;
            });
 
        }
 
        function deletePdf() {
            var data = {
                prm:'D',
                nomePrefix:'req'+RequisicaoService.userLogado.id_usuario+'_'
            };
            RequisicaoService.getPdf(data).then(function(result){
                if (result.status == 'ok') {
                    ok();
                }
            });
        }
 
        function ok() {
            $modalInstance.close(vm.requisicao);
        }
        function cancel(){
        	$modalInstance.dismiss('cancel');
        }
    }
})();
